{% hint style="info" %}
Les Bûches de Chêne sont récupérables au niveau 1 de Bucheron
{% endhint %}

{% columns %}
{% column %}
<figure><img src="../.gitbook/assets/Souche.png" alt=""><figcaption></figcaption></figure>

{% columns %}
